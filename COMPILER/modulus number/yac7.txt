%{
#include<stdio.h>
#include<math.h>
void yyerror (const char *s){
fprintf(stderr,"%s",s);
}
%}
%token NUMBER SQRT POW
%%
Statement:expression {printf("=%d\n",$1);}
expression: expression '+' term {$$=$1+$3;}
           |expression '-' term {$$=$1-$3;}
           |term {$$=$1;}
           ;
term: term '*' factor  {$$=$1*$3;}
     |term '/' factor  {$$=$1/$3;}
     |factor {$$=$1;}
     ;
factor: SQRT '(' expression ')' {$$=sqrt($3);}
      |factor1 {$$=$1;}
      ;
factor1: POW '(' expression ')' {$$=pow($3);}
       | factor2 {$$=$1;}
       ;
factor2: '|' NUMBER '|'
         if($2<0)
         {
              $$=$2*(-1);
          }
        else
         {
            $$=$2;
         }
        |NUMBER {$$=$1;}
        ;
%%
